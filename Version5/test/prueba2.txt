program prueba

	dim a, b, c as matrix[10][10];

	dim ax, ay, bx, by as word;

	dim j, i as word;

	subprocedure matrices

		let ax = 0;
		let ay = 0;
		let bx = 0;
		let by = 0;

		input ("Inserte el tamaño x de la matriz A: ") >> ax;
		input ("Inserte el tamaño y de la matriz A: ") >> ay;
		input ("Inserte el tamaño x de la matriz B: ") >> bx;
		input ("Inserte el tamaño x de la matriz B: ") >> by;

		while ax <> bx || ay <> by do

			print ("ERROR Los tamaños de las matrices son diferentes");
			input ("Inserte el tamaño x de la matriz A: ") >> ax;
			input ("Inserte el tamaño y de la matriz A: ") >> ay;
			input ("Inserte el tamaño x de la matriz B: ") >> bx;
			input ("Inserte el tamaño x de la matriz B: ") >> by;
		whend;

		for i=0 to (ax-1) do
			for j=0 to (ay-1) do
				input ("Inserte valor i j de la matriz A: ") >> a[i][j];
			next;
		next;

		for i=0 to (bx-1) do
			for j=0 to (by-1) do
				input ("Inserte valor i j de la matriz B: ") >> b[i][j];
			next;
		next;
	endsub;

	subprocedure suma

		for i=0 to (ax-1) do
			for j=0 to (ay-1) do
				let c[i][j] = a[i][j] + b[i][j];
			next;
		next;
		print ("Los valores de la matriz resultante son: ");

		for i=0 to (ax-1) do
			for j=0 to (ay-1) do
				print (c[i][j]);
			next;
		next;
	endsub;

	gosub .matrices;
	gosub .suma;
end;
